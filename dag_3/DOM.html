DOM-manipulasjon med JS

På denne siden går vi gjennom noen eksempler av DOM-manipulasjon, og hvordan dette gjøres med litt JavaScript

Knappene kaller på funksjonene beskrevet under seg. 

           function deleteButtons() {
                var buttons = document.getElementsByTagName('button');
                while (buttons.length > 0) {
                    buttons[0].parentNode.removeChild(buttons[0]);
                }
            }
            function buttonClass() {
                var buttons = document.getElementsByTagName('button');
                for (var i = 0; i < buttons.length ; i++) {
                    buttons[i].classList.add("btn", "btn-default");
                }
            }
            function bodyColor(){
                document.getElementsByTagName('body')[0].style="color:red;";
                document.getElementsByTagName('head')[0].appendChild(document.createElement('style')).innerHTML="pre{color:red;}";
            }
            function deleteAll(){
                document.getElementsByTagName('html')[0].innerHTML="";
            }

    function newAlert() {
        window.alert = function (text) { styleAlert(text); }
    }
    function styleAlert(text) {
        var d = document;

        if (d.getElementById('fancyAlert'))
        {
            return;
        }

        var alertBox = d.createElement('div');
        d.getElementsByTagName('body')[0].appendChild(alertBox);
        alertBox.id="fancyAlert";
        alertBox.innerHTML="<div style='width: 100%; background-color: blue; color:white;height:2em; text-align: center; font-size: 1.5em;'>ALERT</div><h2 class='alert alert-info'> Viktig melding! </h2>\n" +"<div class='alert alert-success'>Du skrev: "+text+" </button></div><button style='margin:2em;' class='btn btn-primary btn-lg pull-right' onclick='hideAlert()'>OK</button>";
        alertBox.style="position: fixed; top:10vw; left:40vw; background-color:white; border: 2px solid black; width: 25vw;";
        alertBox.classList;
    }
    function hideAlert() {
        document.getElementById('fancyAlert').parentNode.removeChild(document.getElementById('fancyAlert'));
    }
